
################################################
           excmple目录可以直接使用
    使用本框架需把public设为对外访问目录.
################################################

app = 应用目录
   addons = 模板扩展标签
   controller = 控制器目录
   view = 视图目录
   route.php = 路由文件
config = 配置目录
   config.php = 基础配置文件
extend = 扩展类库目录
public = 对外访问目录
   index.php = 入口文件
   .htaccess = APACHE URL重写
   web.config = IIS URL重写
runtime = 框架运行的缓存/日志等文件
